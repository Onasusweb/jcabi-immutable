48  /**
49   * Map on top of array.
50   *
51   * <p>This class is truly immutable. This means that it never changes
52   * its encapsulated values and is annotated with {@code &#64;Immutable}
53   * annotation.
54   *
55   * @param <K> Map key type
56   * @param <V> Value key type
57   * @author Yegor Bugayenko (yegor@tpc2.com)
58   * @version $Id$
59   * @since 0.1
60   */
61  @Immutable
62  @Loggable(Loggable.DEBUG)
63  @SuppressWarnings({ "rawtypes", "unchecked", "PMD.TooManyMethods" })
64  public final class ArrayMap<K, V> implements ConcurrentMap<K, V> {
65  
66      /**
67       * All entries.
68       */
69      private final transient ImmutableEntry<K, V>[] entries;
70  
71      /**
72       * Public ctor.
73       */
74      public ArrayMap() {
75          this.entries = new ArrayMap.ImmutableEntry[0];
76      }
77  
78      /**
79       * Public ctor.
80       * @param map The original map
81       */
82      @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
83      public ArrayMap(final Map<K, V> map) {
84          if (map == null) {
85              throw new IllegalArgumentException(
86                  "argument of ArrayMap ctor can't be NULL"
87              );
88          }
89          final Set<ArrayMap.ImmutableEntry<K, V>> entrs =
90              new TreeSet<ArrayMap.ImmutableEntry<K, V>>(
91                  new ArrayMap.Cmp<K, V>()
92              );
93          for (final Map.Entry<K, V> entry : map.entrySet()) {
94              entrs.add(new ArrayMap.ImmutableEntry<K, V>(entry));
95          }
96          this.entries = entrs.toArray(new ArrayMap.ImmutableEntry[entrs.size()]);
97      }

99      /**
100      * Make a new one with an extra entry.
101      * @param key The key
102      * @param value The value
103      * @return New map
104      */
105     public ArrayMap<K, V> with(final K key, final V value) {
106         if (key == null) {
107             throw new IllegalArgumentException(
108                 "first argument of ArrayMap#with() can't be NULL"
109             );
110         }
111         if (value == null) {
112             throw new IllegalArgumentException(
113                 "second argument of ArrayMap#with() can't be NULL"
114             );
115         }
116         final ConcurrentMap<K, V> map =
117             new ConcurrentHashMap<K, V>(this.entries.length);
118         map.putAll(this);
119         map.put(key, value);
120         return new ArrayMap<K, V>(map);
121     }

123     /**
124      * Make a new one with these extra entries.
125      * @param ents Entries
126      * @return New map
127      * @since 0.11
128      */
129     public ArrayMap<K, V> with(final Map<K, V> ents) {
130         if (ents == null) {
131             throw new IllegalArgumentException(
132                 "arguments of ArrayMap#with() can't be NULL"
133             );
134         }
135         final ConcurrentMap<K, V> map =
136             new ConcurrentHashMap<K, V>(this.entries.length);
137         map.putAll(this);
138         map.putAll(ents);
139         return new ArrayMap<K, V>(map);
140     }

143      * Make a new one without this key.

148         if (key == null) {
149             throw new IllegalArgumentException(
150                 "argument of ArrayMap#without() can't be NULL"
151             );
152         }
153         final ConcurrentMap<K, V> map =
154             new ConcurrentHashMap<K, V>(this.entries.length);
155         map.putAll(this);
156         map.remove(key);
157         return new ArrayMap<K, V>(map);

160     /**
161      * Make a new one without these keys.
162      * @param keys The keys to remove
163      * @return New map
164      * @since 0.11
165      */
166     public ArrayMap<K, V> without(final Collection<K> keys) {
167         if (keys == null) {
168             throw new IllegalArgumentException(
169                 "arguments of ArrayMap#without() can't be NULL"
170             );
171         }
172         final ConcurrentMap<K, V> map =
173             new ConcurrentHashMap<K, V>(this.entries.length);
174         map.putAll(this);
175         for (final K key : keys) {
176             map.remove(key);
177         }
178         return new ArrayMap<K, V>(map);
179     }

333     /**
334      * Comparator.
335      */
336     private static final class Cmp<K, V> implements
337         Comparator<ArrayMap.ImmutableEntry<K, V>>, Serializable {
338         /**
339          * The Serial version UID.
340          */
341         private static final long serialVersionUID = 4064118000237204080L;
342         @Override
343         public int compare(final ImmutableEntry<K, V> left,
344             final ImmutableEntry<K, V> right) {
345             int compare;
346             if (left.getKey() instanceof Comparable) {
347                 compare = Comparable.class.cast(left.getKey())
348                     .compareTo(right.getKey());
349             } else {
350                 compare = left.getKey().toString()
351                     .compareTo(right.getKey().toString());
352             }
353             return compare;
354         }
355     }

357     /**
358      * Immutable map entry.
359      */
360     @Immutable
361     private static final class ImmutableEntry<K, V> extends
362         AbstractMap.SimpleImmutableEntry<K, V> {
363         /**
364          * Serialization marker.
365          */
366         private static final long serialVersionUID = 1L;
367         /**
368          * Public ctor.
369          * @param entry Entry to encapsulate
370          */
371         private ImmutableEntry(final Map.Entry<K, V> entry) {
372             this(entry.getKey(), entry.getValue());
373         }
374         /**
375          * Public ctor.
376          * @param key The key
377          * @param value The value
378          */
379         private ImmutableEntry(final K key, final V value) {
380             super(key, value);
381         }
382         @Override
383         public String toString() {
384             return String.format("%s=%s", this.getKey(), this.getValue());
385         }
386     }
